def account_find_by_last4(accno_last4):
    
    if(not accno_last4 or accno_last4 is None or accno_last4 ==""):
        return ({"success":False, "Error":"Left_Empty"})
    
    match=[]
    for AccNo in customers:
        if AccNo.endswith(accno_last4):
            match.append(AccNo)

    if(len(match)==0):
        return ({"success":False, "Error":"Not_Found"})
    
    if(len(match)>1):
        return ({"success":False, "Error":"AMBIGUOUS"})
    
    
    return ({"success": True, "account": match[0]})
